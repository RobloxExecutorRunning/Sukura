async function grantGtagItem(titleId, playerOculusId, itemSku) { const baseUrl = `https://${titleId}.playfabapi.com`; try { // 1. "Log in" as the player to get a Session Ticket // This works because GTag copies use the Oculus ID as the Custom ID const loginRes = await fetch(`${baseUrl}/Client/LoginWithCustomID`, { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ TitleId: titleId, CustomId: playerOculusId, CreateAccount: false // Don't create new accounts via the web }) }); const loginData = await loginRes.json(); if (!loginData.data) throw new Error("Player not found or Login disabled."); const sessionTicket = loginData.data.SessionTicket; console.log("Logged in! Session Ticket obtained."); // 2. "Purchase" the free item for the player const purchaseRes = await fetch(`${baseUrl}/Client/PurchaseItem`, { method: 'POST', headers: { 'Content-Type': 'application/json', 'X-Authorization': sessionTicket }, body: JSON.stringify({ ItemId: itemSku, Price: 0, VirtualCurrency: "GR" // Or 'SR' depending on your currency code }) }); const purchaseData = await purchaseRes.json(); if (purchaseData.code === 200) { alert(`Success! Granted ${itemSku} to ${playerOculusId}`); } else { console.error("Purchase Failed:", purchaseData); alert("Error: " + purchaseData.error); } } catch (err) { console.error("Request Failed:", err); alert("Failed to grant item. Check console for details."); } }
